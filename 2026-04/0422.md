## 전체 코드

```jsx
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script crossorigin src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        .item {
            background-color: mistyrose;
            border: 1px dotted black;
            width: 350px;
            display: inline-block;
            margin: 10px;
        }
    </style>
</head>
<body>
    <div id="root"></div>
    
    <script type="text/babel">
        
        const { useState, useEffect } = React;

        function useFetch(url) {

            const [data, setData] = useState(null);  // 서버에서 받아올 유저 목록
            const [loading, setLoading] = useState(true); // 로딩 상태
            const [error, setError] = useState(null); // 에러 상태

            useEffect(() => {

                // 통신 강제 취소 스위치 생성
                const controller = new AbortController();

                // fetch 옵션에 취소 스위치를 연결
                fetch(url, {signal: controller.signal})
                    .then(response => {
                        if (!response.ok) {
                            throw new Error('네트워크 응답 문제가 있습니다.');
                        }
                        return response.json();
                    })
                    .then(responseData => setData(responseData))
                    .catch(error => {
                        if(error.name === 'AbortError') {
                            console.log('통신이 정상적으로 중단되었습니다.');
                        } else {
                            console.error('데이터를 불러오는 데 실패했습니다.', error);
                            setError(error);  // 에러 발생 시 에러 상태에 저장
                        }
                    })
                    .finally(() => setLoading(false));

                    // 클린업 함수: 이 훅을 쓰던 컴포넌트가 화면에서 사라지면 즉시 통신 취소
                    return () => controller.abort();
            }, [url]);

            return { data, loading, error }
        }
 
        function Title() {
            return <h1>회원 목록</h1>;
        }

        function ItemList() {

            const { data: users, loading, error } = useFetch('https://jsonplaceholder.typicode.com/users');

            // Early Return 패턴
            if (loading) {
                return <p>로딩 중...</p>;
            } 
            if (error) {
                return <p>오류가 발생했습니다: {error.message}</p>;
            }
            if (!users) {
                return <h3>데이터가 없습니다.</h3>
            }

            return (
                <div>
                    {users.map(user => <Item key={user.id} user={user}/>)}    
                </div>
            )
        }

        function Item({user}) {
            return (
                <div className="item">  
                    <h4>{user.name}</h4>
                    userId : {user.id} <br/>
                    email : {user.email} <br/>
                    phone : {user.phone} <br/>
                    {/* 옵셔널 체이닝 연산자 사용(에러 방지) */}
                     company name : {user.company?.name} 
                </div>
            )
        }

        function App() {
            return (
                <>
                    <Title />
                    <ItemList />
                </>
            )
        }

        ReactDOM.createRoot(document.getElementById('root')).render(<App />);
        
    </script>
</body>
</html>
```

- `useFetch` : 데이터 가져오는 커스텀 훅
- `Title` : 제목 컴포넌트
- `ItemList` : 데이터 상태 관리 + 리스트 출력
- `Item` : 개별 유저 출력
- `App` : 전체 조합

### MenuAPI

```jsx
import menus from "../data/menu-detail.json"

// 메뉴 데이터 전체 조회
export function getMenuList() {
    return menus;
}

export function getMenuByMenuCode(menuCode) {
    // filter: 배열에서 콜백함수가 true인 요소로만 배열 반환
    // URL 데이터는 문자열이기 때문에 비교를 위해 형변환 해준다.
    return menus.filter(menu => menu.menuCode === parseInt(menuCode))[0];
}

export function searchMenu(searchMenuName) {
    // match() : 포함 여부에 따라 인수 값이 포함되어 있으면 객체 반환
    return menus.filter(menu => menu.menuName.match(searchMenuName))
}
```

```jsx
'use client'
import MenuItem from "@/item/MenuItem";
import { getMenuList } from "@/lib/MenuAPI";
import { useRouter } from "next/navigation";
import { useEffect, useState } from "react";
import boxStyle from "./Menu.module.css";

export default function Menu() {
    
    const [menuList, setMenuList] = useState([]);

    const router = useRouter(); // 라우터 객체 반환 받기
    const [searchValue, setSearchValue] = useState('');
    
    useEffect(() => {
        setMenuList(getMenuList());
    }, []);
    console.log(menuList);

    const onClickHandler = () => {
        // 검색 버튼을 누르면 해당 경로로 이동
        router.push(`/menu/search?menuName=${searchValue}`);
    }

    return (
        <>
            <h1>메뉴 페이지 입니다.</h1>
            <div>
                <input
                    type="search"
                    name="menuName"
                    value={searchValue}
                    onChange={(e) => setSearchValue(e.target.value)}
                />
            </div>
            <button onClick={onClickHandler}>검색</button>
            <div className={boxStyle.MenuBox}>
               {menuList.map(menu => <MenuItem key={menu.menuCode} menu={menu}/>)} 
            </div>
        </>
    )
}
```

---

- usePathname()
    - 현재 URL 경로로 읽어옴 → 활성 탭/메뉴 스타일 적용시
- useParams()
    - [폴더명] 형태의 동적 라우팅 파라미터를 읽어온다 → 상세페이지(/menu/1)
- useSearchParams()
    - URL 끝의 쿼리 스트링(?key=value)를 읽어옴 → 검색 결과 페이지, 필터링
- useRouter()
    - 함수/이벤트 내에서 프로그래밍 방식으로 페이지 이동 → 폼 제출 완료 페이지 이동, 뒤로가기
- ‘use client’ 클라이언트 컴포넌트에서만 사용 가능
