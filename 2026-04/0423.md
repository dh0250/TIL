### APP

- /movie

```jsx
"use client";
import { getMovieList } from "@/api/MovieAPI";
import MovieItem from "@/item/MovieItem";
import { useEffect, useState } from "react";

export default function Movie() {
  const [movieList, setMovieList] = useState();

  useEffect(() => {
    getMovieList().then((movie) => setMovieList(movie));
  }, []);

  console.log(movieList);

  return (
    movieList && (
      <>
        <div className="item">
          {movieList.map((movie) => (
            <MovieItem key={movie} movie={movie} />
          ))}
        </div>
      </>
    )
  );
}

```

- /movie/[movieCd]

```jsx
"use client";
import { getMovieByMovieCode } from "@/api/MovieAPI";
import Link from "next/link";
import { useParams } from "next/navigation";
import { useEffect, useState } from "react";

export default function MovieDetail() {
  const { movieCd } = useParams();
  console.log(movieCd);

  const [movie, setMovie] = useState();

  useEffect(() => {
    //setMovie(getMovieByMovieCode(movieCd));
    const fetchData = async () => {
      const data = await getMovieByMovieCode(movieCd);
      setMovie(data);
    };
    fetchData();
  }, [movieCd]);

  console.log(movie);
  return (
    movie && (
      <>
        <div className="item">
          <h1>{movie.movieNm}</h1>
          <h4>러닝 타임 : {movie.showTm}분</h4>

          <h4>국가: {movie.nations.map((nation) => nation.nations)}</h4>

          <strong>출연</strong>
          <br />
          {movie.actors.map((actor) => {
            return <div key={actor.peopleNm}>{actor.peopleNm}</div>;
          })}

          <Link href="/movie">목록으로</Link>
        </div>
      </>
    )
  );
}

```

## 1️⃣ 공통 레이아웃

- `Header` → 상단
- `Footer` → 하단
- `Layout` → 전체 감싸기

```jsx
<Layout>
  {children}
</Layout>
```

👉 모든 페이지에서 공통 UI 유지

---

## 2️⃣ 페이지 흐름

```
/ (홈)
  ↓
/movie (영화 리스트)
  ↓
/movie/[movieCd] (영화 상세)
```

---

## 3️⃣ 데이터 흐름

- `/movie`
    
    → `getMovieList()` 호출
    
    → 영화 리스트 출력
    
- `/movie/[movieCd]`
    
    → `getMovieByMovieCode(movieCd)` 호출
    
    → 상세 정보 출력
